class Solution
{
    public:
    string reverseWord(string str)
    {
        string s="";
        int len=str.length();
        for(int j=len-1;j>=0;j--)
            s+=str[j];
        return s;
    }
    
};

int main() {
	
	int t;
	cin>>t;
	while(t--)
	{
    	string s;
    	cin >> s;
    	Solution ob;
    	cout << ob.reverseWord(s) << endl;
	}
	return 0;
	
}
