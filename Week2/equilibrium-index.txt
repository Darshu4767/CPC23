class Solution{
    public:
    // Function to find equilibrium point in the array.
    // a: input array
    // n: size of array
    int equilibriumPoint(long long arr[], int n) {
        long long sum=0;
        for(int i=0;i<n;i++)
        {
            sum+=arr[i];
            arr[i]=sum;
        }
        sum=0;
        for(int i=n-1;i>=1;i--)
        {
            if(sum==arr[i-1])
                return i+1;
            sum+=arr[i]-arr[i-1];
        }
        if(sum==0)
            return 1;
        return -1;
    }

};

        cout << ob.equilibriumPoint(a, n) << endl;
